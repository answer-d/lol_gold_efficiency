# riotwatcherの初期化

watcher = RiotWatcher('RGAPI-1c9fdb99-5295-461a-b1e3-45aeddc99ee3')
my_region = 'jp1'

# アイテム一覧の取得（全タグ）
items = watcher.static_data.items(my_region, locale="ja_JP", tags=['all',])

import codecs
